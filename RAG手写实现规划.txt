二、核心流程设计
================

【流程1】文件上传与索引
-----------------------
用户上传文件 
  → RagController.upload()
  → FileUploadService.saveFile()
  → DocumentChunkService.chunkDocument()
     → 按标题分割
     → 按段落分割
     → 添加重叠
  → [对每个分片]
     → EmbeddingService.generateEmbedding()
        → 调用DashScope API
     → VectorStoreService.storeVector()
        → 存储到Milvus
  → 返回上传结果

【流程2】语义检索
-----------------
用户提问
  → RagController.search()
  → RetrievalService.searchSimilarDocuments()
     → EmbeddingService.generateQueryVector()
        → 问题文本向量化
     → VectorStoreService.searchVectors()
        → Milvus相似度搜索
     → 返回TopK结果（含相似度分数）
  → 构建上下文
  → 返回检索结果

```java
package com.example.rag.client;

import com.example.rag.config.RagConfig;
import com.example.rag.exception.EmbeddingException;
import okhttp3.*;
import com.google.gson.*;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Component;

import javax.annotation.PostConstruct;
import java.io.IOException;
import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

@Component
public class DashScopeClient {
    
    private static final Logger logger = LoggerFactory.getLogger(DashScopeClient.class);
    private static final String BASE_URL = "https://dashscope.aliyuncs.com/api/v1";
    private static final String EMBEDDING_ENDPOINT = "/services/embeddings/text-embedding/text-embedding";
    
    private final RagConfig ragConfig;
    private OkHttpClient httpClient;
    private Gson gson;
    
    public DashScopeClient(RagConfig ragConfig) {
        this.ragConfig = ragConfig;
    }
    
    @PostConstruct
    public void init() {
        this.httpClient = new OkHttpClient.Builder()
                .connectTimeout(java.time.Duration.ofSeconds(30))
                .readTimeout(java.time.Duration.ofSeconds(30))
                .build();
        this.gson = new Gson();
        
        if (ragConfig.getApiKey() == null || ragConfig.getApiKey().isEmpty()) {
            throw new IllegalStateException("DashScope API Key未配置");
        }
        
        logger.info("DashScope客户端初始化完成，模型: {}", ragConfig.getEmbeddingModel());
    }
    
    /**
     * 生成单个文本的向量嵌入
     */
    public List<Float> generateEmbedding(String text) throws EmbeddingException {
        if (text == null || text.trim().isEmpty()) {
            throw new EmbeddingException("文本内容不能为空");
        }
        
        try {
            // 构建请求体
            JsonObject requestBody = new JsonObject();
            requestBody.addProperty("model", ragConfig.getEmbeddingModel());
            
            JsonArray inputArray = new JsonArray();
            JsonObject inputObj = new JsonObject();
            inputObj.addProperty("text", text);
            inputArray.add(inputObj);
            requestBody.add("input", inputArray);
            
            // 发送HTTP请求
            RequestBody body = RequestBody.create(
                    gson.toJson(requestBody),
                    MediaType.parse("application/json")
            );
            
            Request request = new Request.Builder()
                    .url(BASE_URL + EMBEDDING_ENDPOINT)
                    .post(body)
                    .addHeader("Authorization", "Bearer " + ragConfig.getApiKey())
                    .addHeader("Content-Type", "application/json")
                    .build();
            
            try (Response response = httpClient.newCall(request).execute()) {
                if (!response.isSuccessful()) {
                    throw new EmbeddingException("API请求失败: " + response.code());
                }
                
                String responseBody = response.body().string();
                return parseEmbeddingResponse(responseBody);
            }
            
        } catch (IOException e) {
            logger.error("调用DashScope API失败", e);
            throw new EmbeddingException("向量生成失败: " + e.getMessage(), e);
        }
    }
    
    /**
     * 批量生成向量嵌入
     */
    public List<List<Float>> generateBatchEmbeddings(List<String> texts) 
            throws EmbeddingException {
        if (texts == null || texts.isEmpty()) {
            return Collections.emptyList();
        }
        
        List<List<Float>> embeddings = new ArrayList<>();
        for (String text : texts) {
            embeddings.add(generateEmbedding(text));
        }
        return embeddings;
    }
    
    /**
     * 解析API响应
     */
    private List<Float> parseEmbeddingResponse(String jsonResponse) 
            throws EmbeddingException {
        try {
            JsonObject json = gson.fromJson(jsonResponse, JsonObject.class);
            
            // 检查是否有错误
            if (json.has("code") && json.get("code").getAsInt() != 0) {
                String message = json.has("message") ? 
                        json.get("message").getAsString() : "未知错误";
                throw new EmbeddingException("API返回错误: " + message);
            }
            
            // 提取向量
            JsonObject output = json.getAsJsonObject("output");
            JsonArray embeddings = output.getAsJsonArray("embeddings");
            
            if (embeddings.isEmpty()) {
                throw new EmbeddingException("API返回空向量");
            }
            
            // 获取第一个文本的向量
            JsonObject firstEmbedding = embeddings.get(0).getAsJsonObject();
            JsonArray vectorArray = firstEmbedding.getAsJsonArray("embedding");
            
            // 转换为List<Float>
            List<Float> vector = new ArrayList<>();
            for (JsonElement element : vectorArray) {
                vector.add(element.getAsFloat());
            }
            
            logger.debug("成功生成向量，维度: {}", vector.size());
            return vector;
            
        } catch (Exception e) {
            throw new EmbeddingException("解析响应失败: " + e.getMessage(), e);
        }
    }
}
```

```java
package com.example.rag.client;

import com.example.rag.config.MilvusConfig;
import com.example.rag.model.DocumentChunk;
import com.example.rag.model.SearchResult;
import io.milvus.client.MilvusServiceClient;
import io.milvus.grpc.DataType;
import io.milvus.param.*;
import io.milvus.param.collection.*;
import io.milvus.param.dml.InsertParam;
import io.milvus.param.dml.SearchParam;
import io.milvus.param.index.CreateIndexParam;
import io.milvus.response.SearchResultsWrapper;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Component;

import javax.annotation.PostConstruct;
import javax.annotation.PreDestroy;
import java.util.*;
import java.util.concurrent.TimeUnit;

@Component
public class MilvusClient {
    
    private static final Logger logger = LoggerFactory.getLogger(MilvusClient.class);
    
    private final MilvusConfig milvusConfig;
    private MilvusServiceClient client;
    
    public MilvusClient(MilvusConfig milvusConfig) {
        this.milvusConfig = milvusConfig;
    }
    
    @PostConstruct
    public void init() {
        try {
            // 连接Milvus
            ConnectParam connectParam = ConnectParam.newBuilder()
                    .withHost(milvusConfig.getHost())
                    .withPort(milvusConfig.getPort())
                    .withConnectTimeout(10, TimeUnit.SECONDS)
                    .build();
            
            this.client = new MilvusServiceClient(connectParam);
            logger.info("成功连接到Milvus: {}", milvusConfig.getAddress());
            
            // 检查并创建Collection
            ensureCollectionExists();
            
        } catch (Exception e) {
            logger.error("初始化Milvus客户端失败", e);
            throw new RuntimeException("Milvus初始化失败", e);
        }
    }
    
    @PreDestroy
    public void destroy() {
        if (client != null) {
            client.close();
            logger.info("Milvus客户端已关闭");
        }
    }
    
    /**
     * 确保Collection存在
     */
    private void ensureCollectionExists() {
        String collectionName = milvusConfig.getCollectionName();
        
        // 检查是否存在
        R<Boolean> hasResponse = client.hasCollection(
                HasCollectionParam.newBuilder()
                        .withCollectionName(collectionName)
                        .build()
        );
        
        if (hasResponse.getStatus() != 0) {
            throw new RuntimeException("检查Collection失败: " + hasResponse.getMessage());
        }
        
        if (!hasResponse.getData()) {
            logger.info("Collection不存在，正在创建: {}", collectionName);
            createCollection(collectionName);
            createIndex(collectionName);
        } else {
            logger.info("Collection已存在: {}", collectionName);
        }
    }
    
    /**
     * 创建Collection
     */
    private void createCollection(String collectionName) {
        // 定义字段
        FieldType idField = FieldType.newBuilder()
                .withName("id")
                .withDataType(DataType.VarChar)
                .withMaxLength(256)
                .withPrimaryKey(true)
                .build();
        
        FieldType vectorField = FieldType.newBuilder()
                .withName("vector")
                .withDataType(DataType.FloatVector)
                .withDimension(milvusConfig.getVectorDimension())
                .build();
        
        FieldType contentField = FieldType.newBuilder()
                .withName("content")
                .withDataType(DataType.VarChar)
                .withMaxLength(8192)
                .build();
        
        FieldType metadataField = FieldType.newBuilder()
                .withName("metadata")
                .withDataType(DataType.JSON)
                .build();
        
        // 创建Schema
        CollectionSchemaParam schema = CollectionSchemaParam.newBuilder()
                .withEnableDynamicField(false)
                .addFieldType(idField)
                .addFieldType(vectorField)
                .addFieldType(contentField)
                .addFieldType(metadataField)
                .build();
        
        // 创建Collection
        CreateCollectionParam createParam = CreateCollectionParam.newBuilder()
                .withCollectionName(collectionName)
                .withDescription("RAG文档向量集合")
                .withSchema(schema)
                .withShardsNum(2)
                .build();
        
        R<RpcStatus> response = client.createCollection(createParam);
        if (response.getStatus() != 0) {
            throw new RuntimeException("创建Collection失败: " + response.getMessage());
        }
        
        logger.info("Collection创建成功: {}", collectionName);
    }
    
    /**
     * 创建索引
     */
    private void createIndex(String collectionName) {
        CreateIndexParam indexParam = CreateIndexParam.newBuilder()
                .withCollectionName(collectionName)
                .withFieldName("vector")
                .withIndexType(IndexType.IVF_FLAT)
                .withMetricType(MetricType.L2)
                .withExtraParam("{\"nlist\":" + milvusConfig.getNlist() + "}")
                .build();
        
        R<RpcStatus> response = client.createIndex(indexParam);
        if (response.getStatus() != 0) {
            throw new RuntimeException("创建索引失败: " + response.getMessage());
        }
        
        logger.info("索引创建成功");
    }
    
    /**
     * 插入向量数据
     */
    public void insertVector(DocumentChunk chunk, List<Float> vector) {
        try {
            // 加载Collection
            client.loadCollection(
                    LoadCollectionParam.newBuilder()
                            .withCollectionName(milvusConfig.getCollectionName())
                            .build()
            );
            
            // 构建元数据
            Map<String, Object> metadata = new HashMap<>();
            metadata.put("sourceFile", chunk.getSourceFile());
            metadata.put("chunkIndex", chunk.getChunkIndex());
            if (chunk.getTitle() != null) {
                metadata.put("title", chunk.getTitle());
            }
            
            // 构建字段数据
            List<InsertParam.Field> fields = new ArrayList<>();
            fields.add(new InsertParam.Field("id", 
                    Collections.singletonList(chunk.getId())));
            fields.add(new InsertParam.Field("vector", 
                    Collections.singletonList(vector)));
            fields.add(new InsertParam.Field("content", 
                    Collections.singletonList(chunk.getContent())));
            
            // 将metadata转为JSON字符串
            com.google.gson.Gson gson = new com.google.gson.Gson();
            String metadataJson = gson.toJson(metadata);
            fields.add(new InsertParam.Field("metadata", 
                    Collections.singletonList(metadataJson)));
            
            // 插入数据
            InsertParam insertParam = InsertParam.newBuilder()
                    .withCollectionName(milvusConfig.getCollectionName())
                    .withFields(fields)
                    .build();
            
            R<MutationResult> response = client.insert(insertParam);
            if (response.getStatus() != 0) {
                throw new RuntimeException("插入数据失败: " + response.getMessage());
            }
            
            logger.debug("向量插入成功: id={}", chunk.getId());
            
        } catch (Exception e) {
            logger.error("插入向量失败", e);
            throw new RuntimeException("插入向量失败", e);
        }
    }
    
    /**
     * 搜索相似向量
     */
    public List<SearchResult> searchSimilarVectors(List<Float> queryVector, int topK) {
        try {
            // 加载Collection
            client.loadCollection(
                    LoadCollectionParam.newBuilder()
                            .withCollectionName(milvusConfig.getCollectionName())
                            .build()
            );
            
            // 构建搜索参数
            SearchParam searchParam = SearchParam.newBuilder()
                    .withCollectionName(milvusConfig.getCollectionName())
                    .withVectorFieldName("vector")
                    .withVectors(Collections.singletonList(queryVector))
                    .withTopK(topK)
                    .withMetricType(MetricType.L2)
                    .withOutFields(Arrays.asList("id", "content", "metadata"))
                    .withParams("{\"nprobe\":10}")
                    .build();
            
            // 执行搜索
            R<SearchResults> response = client.search(searchParam);
            if (response.getStatus() != 0) {
                throw new RuntimeException("搜索失败: " + response.getMessage());
            }
            
            // 解析结果
            SearchResultsWrapper wrapper = new SearchResultsWrapper(
                    response.getData().getResults()
            );
            
            List<SearchResult> results = new ArrayList<>();
            com.google.gson.Gson gson = new com.google.gson.Gson();
            
            for (int i = 0; i < wrapper.getRowRecords(0).size(); i++) {
                String id = (String) wrapper.getIDScore(0).get(i).get("id");
                String content = (String) wrapper.getFieldData("content", 0).get(i);
                float score = wrapper.getIDScore(0).get(i).getScore();
                String metadataStr = (String) wrapper.getFieldData("metadata", 0).get(i);
                
                // 解析元数据
                Map<String, Object> metadata = gson.fromJson(metadataStr, Map.class);
                
                SearchResult result = new SearchResult(
                        id, content, score, 
                        (String) metadata.get("sourceFile"),
                        metadata
                );
                results.add(result);
            }
            
            logger.info("搜索完成，找到{}个结果", results.size());
            return results;
            
        } catch (Exception e) {
            logger.error("搜索向量失败", e);
            throw new RuntimeException("搜索向量失败", e);
        }
    }
    
    /**
     * 删除指定文件的所有向量
     */
    public void deleteBySourceFile(String sourceFile) {
        try {
            client.loadCollection(
                    LoadCollectionParam.newBuilder()
                            .withCollectionName(milvusConfig.getCollectionName())
                            .build()
            );
            
            // 构建删除表达式
            String expr = String.format("metadata[\"sourceFile\"] == \"%s\"", sourceFile);
            
            DeleteParam deleteParam = DeleteParam.newBuilder()
                    .withCollectionName(milvusConfig.getCollectionName())
                    .withExpr(expr)
                    .build();
            
            R<MutationResult> response = client.delete(deleteParam);
            if (response.getStatus() != 0) {
                logger.warn("删除数据警告: {}", response.getMessage());
            } else {
                logger.info("删除文件向量成功: {}, 删除{}条记录", 
                        sourceFile, response.getData().getDeleteCnt());
            }
            
        } catch (Exception e) {
            logger.warn("删除文件向量失败（可能不存在）: {}", e.getMessage());
        }
    }
}
```

```java
package com.example.rag.service;

import com.example.rag.config.RagConfig;
import com.example.rag.model.UploadResult;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;

import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.nio.file.Paths;
import java.util.Arrays;
import java.util.List;

@Service
public class FileUploadService {
    
    private static final Logger logger = LoggerFactory.getLogger(FileUploadService.class);
    
    private final RagConfig ragConfig;
    
    public FileUploadService(RagConfig ragConfig) {
        this.ragConfig = ragConfig;
    }
    
    /**
     * 保存上传的文件
     */
    public UploadResult saveFile(MultipartFile file) {
        try {
            // 1. 校验文件
            validateFile(file);
            
            // 2. 创建上传目录
            Path uploadDir = Paths.get(ragConfig.getUploadDir()).normalize();
            if (!Files.exists(uploadDir)) {
                Files.createDirectories(uploadDir);
            }
            
            // 3. 保存文件
            String originalFilename = file.getOriginalFilename();
            Path filePath = uploadDir.resolve(originalFilename).normalize();
            
            // 如果文件已存在，先删除
            if (Files.exists(filePath)) {
                logger.info("文件已存在，将覆盖: {}", filePath);
                Files.delete(filePath);
            }
            
            Files.copy(file.getInputStream(), filePath);
            
            logger.info("文件上传成功: {}", filePath);
            
            return UploadResult.success(
                    originalFilename,
                    filePath.toString(),
                    file.getSize(),
                    0  // 分片数量由后续步骤填充
            );
            
        } catch (IOException e) {
            logger.error("文件上传失败", e);
            return UploadResult.failure(
                    file.getOriginalFilename(),
                    Arrays.asList("文件上传失败: " + e.getMessage())
            );
        }
    }
    
    /**
     * 校验文件
     */
    private void validateFile(MultipartFile file) {
        if (file.isEmpty()) {
            throw new IllegalArgumentException("文件不能为空");
        }
        
        String filename = file.getOriginalFilename();
        if (filename == null || filename.isEmpty()) {
            throw new IllegalArgumentException("文件名不能为空");
        }
        
        // 检查扩展名
        String extension = getFileExtension(filename);
        List<String> allowedExts = Arrays.asList(
                ragConfig.getAllowedExtensions().split(",")
        );
        
        if (!allowedExts.contains(extension.toLowerCase())) {
            throw new IllegalArgumentException(
                    "不支持的文件格式: " + extension + 
                    "，仅支持: " + ragConfig.getAllowedExtensions()
            );
        }
    }
    
    private String getFileExtension(String filename) {
        int lastIndexOf = filename.lastIndexOf(".");
        if (lastIndexOf == -1) {
            return "";
        }
        return filename.substring(lastIndexOf + 1).toLowerCase();
    }
}
```

```java
package com.example.rag.service;

import com.example.rag.config.RagConfig;
import com.example.rag.model.DocumentChunk;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Service;

import java.util.ArrayList;
import java.util.List;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

@Service
public class DocumentChunkService {
    
    private static final Logger logger = LoggerFactory.getLogger(DocumentChunkService.class);
    
    private final RagConfig ragConfig;
    
    public DocumentChunkService(RagConfig ragConfig) {
        this.ragConfig = ragConfig;
    }
    
    /**
     * 智能分片文档
     */
    public List<DocumentChunk> chunkDocument(String content, String sourceFile) {
        List<DocumentChunk> chunks = new ArrayList<>();
        
        if (content == null || content.trim().isEmpty()) {
            logger.warn("文档内容为空: {}", sourceFile);
            return chunks;
        }
        
        // 1. 按标题分割
        List<Section> sections = splitByHeadings(content);
        
        // 2. 对每个章节进一步分片
        int globalChunkIndex = 0;
        for (Section section : sections) {
            List<DocumentChunk> sectionChunks = chunkSection(section, sourceFile, globalChunkIndex);
            chunks.addAll(sectionChunks);
            globalChunkIndex += sectionChunks.size();
        }
        
        logger.info("文档分片完成: {} -> {} 个分片", sourceFile, chunks.size());
        return chunks;
    }
    
    /**
     * 按Markdown标题分割
     */
    private List<Section> splitByHeadings(String content) {
        List<Section> sections = new ArrayList<>();
        
        Pattern headingPattern = Pattern.compile("^(#{1,6})\\s+(.+)$", Pattern.MULTILINE);
        Matcher matcher = headingPattern.matcher(content);
        
        int lastEnd = 0;
        String currentTitle = null;
        
        while (matcher.find()) {
            if (lastEnd < matcher.start()) {
                String sectionContent = content.substring(lastEnd, matcher.start()).trim();
                if (!sectionContent.isEmpty()) {
                    sections.add(new Section(currentTitle, sectionContent, lastEnd));
                }
            }
            currentTitle = matcher.group(2).trim();
            lastEnd = matcher.start();
        }
        
        // 添加最后一个章节
        if (lastEnd < content.length()) {
            String sectionContent = content.substring(lastEnd).trim();
            if (!sectionContent.isEmpty()) {
                sections.add(new Section(currentTitle, sectionContent, lastEnd));
            }
        }
        
        // 如果没有标题，整个文档作为一个章节
        if (sections.isEmpty()) {
            sections.add(new Section(null, content, 0));
        }
        
        return sections;
    }
    
    /**
     * 对章节进行分片
     */
    private List<DocumentChunk> chunkSection(Section section, String sourceFile, 
                                            int startChunkIndex) {
        List<DocumentChunk> chunks = new ArrayList<>();
        String content = section.content;
        String title = section.title;
        
        // 如果章节内容小于最大尺寸，直接作为一个分片
        if (content.length() <= ragConfig.getChunkMaxSize()) {
            DocumentChunk chunk = new DocumentChunk(content, sourceFile, startChunkIndex);
            chunk.setTitle(title);
            chunks.add(chunk);
            return chunks;
        }
        
        // 按段落分割
        List<String> paragraphs = splitByParagraphs(content);
        
        StringBuilder currentChunk = new StringBuilder();
        int chunkIndex = startChunkIndex;
        
        for (String paragraph : paragraphs) {
            if (currentChunk.length() > 0 && 
                currentChunk.length() + paragraph.length() > ragConfig.getChunkMaxSize()) {
                
                // 保存当前分片
                String chunkContent = currentChunk.toString().trim();
                DocumentChunk chunk = new DocumentChunk(chunkContent, sourceFile, chunkIndex++);
                chunk.setTitle(title);
                chunks.add(chunk);
                
                // 开始新分片，包含重叠
                String overlap = getOverlapText(chunkContent);
                currentChunk = new StringBuilder(overlap);
            }
            
            currentChunk.append(paragraph).append("\n\n");
        }
        
        // 保存最后一个分片
        if (currentChunk.length() > 0) {
            String chunkContent = currentChunk.toString().trim();
            DocumentChunk chunk = new DocumentChunk(chunkContent, sourceFile, chunkIndex);
            chunk.setTitle(title);
            chunks.add(chunk);
        }
        
        return chunks;
    }
    
    /**
     * 按段落分割
     */
    private List<String> splitByParagraphs(String content) {
        List<String> paragraphs = new ArrayList<>();
        String[] parts = content.split("\n\n+");
        for (String part : parts) {
            String trimmed = part.trim();
            if (!trimmed.isEmpty()) {
                paragraphs.add(trimmed);
            }
        }
        return paragraphs;
    }
    
    /**
     * 获取重叠文本
     */
    private String getOverlapText(String text) {
        int overlapSize = Math.min(ragConfig.getChunkOverlap(), text.length());
        if (overlapSize <= 0) {
            return "";
        }
        
        String overlap = text.substring(text.length() - overlapSize);
        
        // 尝试在句子边界截断
        int lastSentenceEnd = Math.max(
                overlap.lastIndexOf('。'),
                Math.max(overlap.lastIndexOf('？'), overlap.lastIndexOf('！'))
        );
        
        if (lastSentenceEnd > overlapSize / 2) {
            return overlap.substring(lastSentenceEnd + 1).trim();
        }
        
        return overlap.trim();
    }
    
    /**
     * 章节数据类
     */
    private static class Section {
        String title;
        String content;
        int startIndex;
        
        Section(String title, String content, int startIndex) {
            this.title = title;
            this.content = content;
            this.startIndex = startIndex;
        }
    }
}
```

```java
package com.example.rag.service;

import com.example.rag.client.MilvusClient;
import com.example.rag.model.DocumentChunk;
import com.example.rag.model.SearchResult;
import org.slf4j.Logger;

import org.springframework.stereotype.Service;

@Service
public class VectorStoreService {
    
    private static final Logger logger = LoggerFactory.getLogger(VectorStoreService.class);
    
    private final MilvusClient milvusClient;
    private final EmbeddingService embeddingService;
    
    public VectorStoreService(MilvusClient milvusClient, EmbeddingService embeddingService) {
        this.milvusClient = milvusClient;
        this.embeddingService = embeddingService;
    }
    
    /**
     * 存储文档分片的向量
     */
    public void storeVector(DocumentChunk chunk) {
        try {
            logger.info("开始存储向量: id={}, source={}", chunk.getId(), chunk.getSourceFile());
            
            // 1. 生成向量
            List<Float> vector = embeddingService.generateEmbedding(chunk.getContent());
            
            // 2. 存储到Milvus
            milvusClient.insertVector(chunk, vector);
            
            logger.info("向量存储成功: id={}", chunk.getId());
            
        } catch (Exception e) {
            logger.error("存储向量失败: id={}", chunk.getId(), e);
            throw new RuntimeException("存储向量失败", e);
        }
    }
    
    /**
     * 批量存储向量
     */
    public void storeVectors(List<DocumentChunk> chunks) {
        logger.info("开始批量存储向量，数量: {}", chunks.size());
        
        for (int i = 0; i < chunks.size(); i++) {
            DocumentChunk chunk = chunks.get(i);
            try {
                storeVector(chunk);
                logger.info("进度: {}/{}", i + 1, chunks.size());
            } catch (Exception e) {
                logger.error("第{}个分片存储失败", i + 1, e);
                throw e;
            }
        }
        
        logger.info("批量存储完成");
    }
    
    /**
     * 搜索相似文档
     */
    public List<SearchResult> searchSimilarDocuments(String query, int topK) {
        try {
            logger.info("开始搜索相似文档: query={}, topK={}", query, topK);
            
            // 1. 生成查询向量
            List<Float> queryVector = embeddingService.generateQueryVector(query);
            
            // 2. 搜索Milvus
            List<SearchResult> results = milvusClient.searchSimilarVectors(queryVector, topK);
            
            logger.info("搜索完成，找到{}个结果", results.size());
            return results;
            
        } catch (Exception e) {
            logger.error("搜索失败", e);
            throw new RuntimeException("搜索失败", e);
        }
    }
    
    /**
     * 删除文件的向量数据
     */
    public void deleteBySourceFile(String sourceFile) {
        logger.info("删除文件向量: {}", sourceFile);
        milvusClient.deleteBySourceFile(sourceFile);
    }
```

```java
package com.example.rag.service;

import com.example.rag.config.RagConfig;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.stream.Collectors;

@Service
public class RetrievalService {
    
    private static final Logger logger = LoggerFactory.getLogger(RetrievalService.class);
    
    private final VectorStoreService vectorStoreService;
    private final RagConfig ragConfig;
    
    public RetrievalService(VectorStoreService vectorStoreService, RagConfig ragConfig) {
        this.vectorStoreService = vectorStoreService;
        this.ragConfig = ragConfig;
    }
    
    /**
     * 检索相关文档
     */
    public List<SearchResult> retrieve(String query) {
        logger.info("开始检索: {}", query);
        
        // 1. 搜索相似文档
        List<SearchResult> results = vectorStoreService.searchSimilarDocuments(
                query, 
                ragConfig.getTopK()
        );
        
        // 2. 过滤低相似度结果
        List<SearchResult> filteredResults = results.stream()
                .filter(r -> r.getScore() <= ragConfig.getSimilarityThreshold())
                .collect(Collectors.toList());
        
        logger.info("检索完成，返回{}个结果（过滤后）", filteredResults.size());
        return filteredResults;
    }
    
    /**
     * 构建上下文
     */
    public String buildContext(List<SearchResult> results) {
        StringBuilder context = new StringBuilder();
        
        for (int i = 0; i < results.size(); i++) {
            SearchResult result = results.get(i);
            context.append("【参考资料 ").append(i + 1).append("】\n");
            context.append("来源: ").append(result.getSourceFile()).append("\n");
            context.append("内容: ").append(result.getContent()).append("\n\n");
        }
        
        return context.toString();
    }
}
```

```java
@RequestMapping("/api/rag")
public class RagController {

    private final DocumentChunkService documentChunkService;
    private final VectorStoreService vectorStoreService;
    private final RetrievalService retrievalService;
    
    public RagController(FileUploadService fileUploadService,
                        DocumentChunkService documentChunkService,
                        VectorStoreService vectorStoreService,
                        RetrievalService retrievalService) {
        this.fileUploadService = fileUploadService;
        this.documentChunkService = documentChunkService;
        this.vectorStoreService = vectorStoreService;
        this.retrievalService = retrievalService;
    }
    
    /**
     * 文件上传接口
     */
    @PostMapping(value = "/upload", consumes = "multipart/form-data")
    public ResponseEntity<Map<String, Object>> upload(@RequestParam("file") MultipartFile file) {
        Map<String, Object> response = new HashMap<>();
        
        try {
            // 1. 保存文件
            UploadResult uploadResult = fileUploadService.saveFile(file);
            
            if (!uploadResult.isSuccess()) {
                response.put("success", false);
                response.put("errors", uploadResult.getErrors());
                return ResponseEntity.badRequest().body(response);
            }
            
            // 2. 读取文件内容
            String content = Files.readString(Paths.get(uploadResult.getFilePath()));
            
            // 3. 删除旧向量数据（去重）
            vectorStoreService.deleteBySourceFile(uploadResult.getFilePath());
            
            // 4. 文档分片
            List<DocumentChunk> chunks = documentChunkService.chunkDocument(
                    content, 
                    uploadResult.getFilePath()
            );
            
            // 5. 存储向量
            vectorStoreService.storeVectors(chunks);
            
            // 6. 更新上传结果
            uploadResult.setChunkCount(chunks.size());
            
            response.put("success", true);
            response.put("data", uploadResult);
            return ResponseEntity.ok(response);
            
        } catch (Exception e) {
            response.put("success", false);
            response.put("error", e.getMessage());
            return ResponseEntity.internalServerError().body(response);
        }
    }
    
    /**

     */

            if (query == null || query.trim().isEmpty()) {
                response.put("success", false);
                response.put("error", "查询词不能为空");
                return ResponseEntity.badRequest().body(response);
            }
            
            // 检索相关文档
            List<SearchResult> results = retrievalService.retrieve(query);
            
            // 构建上下文
            String context = retrievalService.buildContext(results);
            
            response.put("success", true);
            response.put("results", results);
            response.put("context", context);
            response.put("count", results.size());
            return ResponseEntity.ok(response);
            
        } catch (Exception e) {
            response.put("success", false);
            response.put("error", e.getMessage());
            return ResponseEntity.internalServerError().body(response);
        }
    }
```

七、关键设计要点
================
